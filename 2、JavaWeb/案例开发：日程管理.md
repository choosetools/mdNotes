        //账号框验证
        function checkUsername(){
            var regUserName = /^[a-zA-Z0-9]{5,10}$/;
            var unInput = document.getElementById("usernameInput");
            var unMsg = document.getElementById("usernameMsg")
            if(!regUserName.test(unInput.value)){
                unMsg.innerText = "格式有误";
                return false;
            }
            unMsg.innerText = "";
            return true;
        }

        //密码框验证
        function checkUserPwd(){
            var regPwd = /^\d{6}$/;
            var upInput = document.getElementById("userPwdInput");
            var upMsg = document.getElementById("userPwdMsg")
            if(!regPwd.test(upInput.value)){
                upMsg.innerText = "格式有误";
                return false;
            }
            upMsg.innerText = "";
            return true;
        }

        //确认密码框验证
        function checkReUserPwd(){
            //密码框元素
            var pwdInput = document.getElementById("userPwdInput");
            //确认密码框元素
            var rePwdInput = document.getElementById("reUserPwdInput");

            var rePwdMsg = document.getElementById("reUserPwdMsg");

            if(rePwdInput.value != pwdInput.value){
                rePwdMsg.innerText = "两次输入密码不一致";
                return false;
            }
            rePwdMsg.innerText = "";
            return true;
        }

        //提交验证
        function checkForm(){
            return checkUsername() && checkReUserPwd() && checkUserPwd();
        }

        //重置清除所有提示
        function clearMsg(){
            var unMsg = document.getElementById("usernameMsg")
            var upMsg = document.getElementById("userPwdMsg")
            var rePwdMsg = document.getElementById("reUserPwdMsg");
            unMsg.innerText = "";
            upMsg.innerText = "";
            rePwdMsg.innerText = "";
        }

    </script>
